import java.util.Scanner;
public class Number {
	public static void
	guessingNumberGame()
	{
		Scanner sc = new Scanner(System.in);
		int number = 1 + (int)(100
						 * Math.random());
		int K = 8;
		int i, guess;
		System.out.println("A number is generate between 1 to 100.Guess the number within 8 trials.");
		for (i = 0; i < K; i++) {
			System.out.println(
				" Guess the number:");
			guess = sc.nextInt();
			if (number == guess) {
				System.out.println(
					"Congratulations! You have guessed the number.");
				break;
			}
			else if (number > guess
					&& i != K - 1) {
				System.out.println(
					"The number is too high compare to " + guess);
			}
			else if (number < guess
					&& i != K - 1) {
				System.out.println(
					"The number is too low compare to " + guess);
			}
		}
		if (i == K) {
			System.out.println(
				"You have exhausted 8 trials.");
			System.out.println(
				"The number was " + number);
		}
	}
	public static void
	main(String arg[])
	{
		guessingNumberGame();
	}
}
